Get shows by day of the week, time descending
SELECT * FROM shows WHERE day_of_week = ['Tuesday'] ORDER BY show_time DESC;

Get a episode's playlist by its ID
SELECT songs.title, songs.artist, songs.album, songs.album_link, songs.album_image, playlist_songs.song_order 
FROM playlists 
RIGHT JOIN playlist_songs ON playlists.id = playlist_songs.playlist_id 
RIGHT JOIN songs ON playlist_songs.song_id = songs.id 
WHERE playlists.id = [1] 
ORDER BY playlist_songs.song_order;

Get all playlists for a DJ based on their ID
SELECT playlists.id, playlists.date, playlists.description 
FROM members 
RIGHT JOIN shows ON members.id = shows.dj_id 
RIGHT JOIN show_playlists ON shows.id = show_playlists.show_id 
RIGHT JOIN playlists ON show_playlists.playlist_id = playlists.id 
WHERE members.id = [1] 
ORDER BY playlists.date DESC;

Get all DJ ids
SELECT id 
FROM members 
WHERE is_dj;

Get the most recent episode of a show based on its ID
SELECT playlists.id, playlists.date, playlists.description 
FROM shows 
RIGHT JOIN show_playlists ON shows.id = show_playlists.show_id 
RIGHT JOIN playlists ON show_playlists.playlist_id = playlists.id 
WHERE shows.id = [1] 
ORDER BY playlists.date DESC;

Get all of a member's favorite shows by ID
SELECT show_id 
FROM member_favorites 
WHERE member_id = [1];
